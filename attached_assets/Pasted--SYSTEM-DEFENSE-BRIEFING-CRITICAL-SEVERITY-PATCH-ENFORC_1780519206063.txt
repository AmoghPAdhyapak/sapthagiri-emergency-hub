// Central Shared Data Memory Pools
const DB = global.DB || {
  users: new Map(),           // Authentication reference map
  patientsFolder: new Map(),  // Central Patients Folder shared with Staff Portal
  encounters: new Map()       // Shared triage logs map
};
global.DB = DB;

// Unified Registration Endpoint resolving Failure #1
app.post('/api/auth/patient-signup', (req, res) => {
  const { fullName, phone, email, age, password } = req.body;

  if (!fullName || !phone || !password) {
    return res.status(400).json({ success: false, error: "Missing required onboarding parameters." });
  }

  // Check if credentials exist
  const existingUser = Array.from(DB.users.values()).find(u => u.phone === phone);
  if (existingUser) {
    return res.status(400).json({ success: false, error: "An account with this primary phone number is already registered." });
  }

  const generatedPatientId = `Patient-${Math.floor(1000 + Math.random() * 9000)}`;
  const saltSessionRef = `SESSION-REF-${Math.random().toString(36).substring(7).toUpperCase()}`;

  const accountRecord = {
    userId: generatedPatientId,
    role: "patient",
    name: fullName,
    phone,
    email: email || "Not Provided",
    age: parseInt(age) || "Unknown",
    password, 
    sessionRef: saltSessionRef
  };

  // Push simultaneously to the authentication table AND the shared staff directory folder
  DB.users.set(generatedPatientId, accountRecord);
  DB.patientsFolder.set(generatedPatientId, {
    patientId: generatedPatientId,
    name: fullName,
    phone,
    email: email || "Not Provided",
    age: parseInt(age) || "Unknown",
    medicalHistory: [],
    treatmentTimeline: []
  });

  res.status(200).json({ 
    success: true, 
    message: "Registration successful. Portal sync established.", 
    user: { userId: generatedPatientId, role: "patient", patientId: generatedPatientId } 
  });
});

// Patients Directory Query for Staff Portal -> Patients Folder view
app.get('/api/patients/all', (req, res) => {
  const dataset = Array.from(DB.patientsFolder.values());
  res.status(200).json({ success: true, patients: dataset });
});

APPEND-ONLY TIMELINE MATRIX FOR CONTINUITY (POST /api/patients/append-continuity)

Implement a chronological, append-only history engine that blocks overwriting past files to resolve Failure #2:

```javascript
app.post('/api/patients/append-continuity', (req, res) => {
  const { patientId, doctorName, doctorId, hospitalName, medRegId, phone, treatmentNotes } = req.body;

  if (!patientId || !doctorName || !medRegId || !treatmentNotes) {
    return res.status(400).json({ success: false, error: "All clinical tracking validations are mandatory." });
  }

  const targetedRecord = DB.patientsFolder.get(patientId);
  if (!targetedRecord) {
    return res.status(404).json({ success: false, error: "Target Patient File could not be retrieved from master folder registry." });
  }

  const newContinuityLogNode = {
    logId: `LOG-EXT-${Date.now()}`,
    doctorName,
    doctorId: doctorId || "N/A",
    hospitalName,
    medicalRegistrationId: medRegId,
    phoneNumber: phone || "Not Provided",
    clinicalObservations: treatmentNotes,
    timestamp: new Date().toLocaleDateString('en-GB', { day: 'numeric', month: 'short', year: 'numeric', hour: '2-digit', minute: '2-digit' })
  };

  // Safe Array Stacking Injection — Appends chronologically without modifying historical records
  if (!targetedRecord.treatmentTimeline) targetedRecord.treatmentTimeline = [];
  targetedRecord.treatmentTimeline.push(newContinuityLogNode);
  
  DB.patientsFolder.set(patientId, targetedRecord);
  res.status(200).json({ success: true, updatedTimeline: targetedRecord.treatmentTimeline });
});
```
