      ...dataPacket
    };

    if (!patientFile.forensicLifecycleTimeline) {
      patientFile.forensicLifecycleTimeline = [];

    // Un-deletable chronological accumulation

    this.patientsFolder.set(patientId, patientFile);
    
    // Lock memory matrix persistently inside localStorage

app.post('/api/triage/commit-status', (req, res) => {
  const { patientId, doctorId, doctorName, requestedStatus, requestedVerdict, textInputs } = req.body;
  
  let finalVerdict = requestedVerdict; // Initial doctor designation: GREEN | YELLOW
  let aiOverrideTriggered = false;
  let overrideReasonString = "";
  
  const textContentToScan = `${textInputs.symptoms || ""} ${textInputs.reason || ""} ${textInputs.observations || ""}`.toLowerCase();
  
  // Critical Severity Keywords Scan Logic
  const criticalPhrases = ["heart attack", "cardiac arrest", "breathing collapse", "internal bleeding", "unconscious", "stroke"];
  const matchedPhrase = criticalPhrases.find(phrase => textContentToScan.includes(phrase));

  if (matchedPhrase) {
    finalVerdict = "RED"; // Hard override to RED Emergency Hub
    aiOverrideTriggered = true;
    
    // Frame Contextual Explanation Banners matching LIDC criteria
    if (matchedPhrase.includes("heart") || matchedPhrase.includes("cardiac")) {
      overrideReasonString = "AI Override Activated: Cardiac emergency indicators detected.";
    } else if (matchedPhrase.includes("breathing") || matchedPhrase.includes("collapse")) {
      overrideReasonString = "Critical respiratory failure language detected.";
    } else {
      overrideReasonString = "Neurological emergency severity threshold exceeded.";
    }
  }

  // Handle Forensic Appends based on clinical status
  if (aiOverrideTriggered) {
    GlobalHospitalDatabase.appendForensicLog(patientId, "AI_OVERRIDE", {
      originalDoctorVerdict: requestedVerdict,
      aiOverrideReason: overrideReasonString,
      detectedEmergencyPhrase: matchedPhrase,
      authorizingDoctorId: doctorId
    });
  }

  // Write base metrics to timeline history logs
  GlobalHospitalDatabase.appendForensicLog(patientId, "DOCTOR_CHAIN", {
    doctorName,
    doctorId,
    roleInTreatment: `Status Mutation to ${requestedStatus}`,
    escalationDecision: finalVerdict,
    observationNotes: textInputs.observations
  });

  // If case is flagged deceased, append specialized report model structure
  if (requestedStatus === "DECEASED") {
    GlobalHospitalDatabase.appendForensicLog(patientId, "DEATH_REPORT", {
      exactCauseOfDeath: textInputs.reason || "Undetermined",
      symptomsBeforeDeath: textInputs.symptoms,
      attendingDoctors: [doctorName],
      finalClinicalNotes: textInputs.observations
    });
  }

  return res.status(200).json({
    success: true,
    aiOverrideActive: aiOverrideTriggered,
    bannerReason: overrideReasonString,
    finalVerdictTier: finalVerdict
  });
});
