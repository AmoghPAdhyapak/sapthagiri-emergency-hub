SERVER DATABASE INITIALIZATION:

const express = require('express');
const sqlite3 = require('sqlite3').verbose();
const app = express();
app.use(express.json());

const db = new sqlite3.Database('./hospital_ecosystem.db', (err) => {
  if (err) console.error('FAIL: SQLite disk binding initialization error:', err.message);
  else console.log('SUCCESS: Persistent SQL database successfully locked to file system storage.');
});

db.serialize(() => {
  db.run(`CREATE TABLE IF NOT EXISTS STAFF_ACCOUNTS (
    doctorId TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    role TEXT NOT NULL,
    passwordHash TEXT NOT NULL,
    createdAt TEXT NOT NULL
  )`);

  db.run(`CREATE TABLE IF NOT EXISTS PATIENT_MASTER_FOLDER (
    patientId TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    age INTEGER NOT NULL,
    allergies TEXT,
    phone TEXT NOT NULL,
    survivalState TEXT DEFAULT 'Stable',
    createdAt TEXT NOT NULL
  )`);

  db.run(`CREATE TABLE IF NOT EXISTS ENCOUNTERS (
    encounterId TEXT PRIMARY KEY,
    patientId TEXT NOT NULL,
    symptoms TEXT,
    verdict TEXT NOT NULL,
    assignedDoctor TEXT,
    status TEXT NOT NULL
  )`);

  db.run(`CREATE TABLE IF NOT EXISTS FORENSIC_LIFECYCLE_LOGS (
    logId TEXT PRIMARY KEY,
    patientId TEXT NOT NULL,
    eventType TEXT NOT NULL,
    timestamp TEXT NOT NULL,
    forensicPayload TEXT
  )`);
});

UNIFIED SYNC API

Use unified SQL synchronization route:

app.get('/api/patients/sync-master/:patientId', (req, res) => {
  const { patientId } = req.params;
  
  db.get(`SELECT * FROM PATIENT_MASTER_FOLDER WHERE patientId = ?`, [patientId], (err, profile) => {
    if (!profile) return res.status(404).json({ success: false, error: "Patient record completely missing from persistent storage rows." });
    
    db.all(`SELECT * FROM FORENSIC_LIFECYCLE_LOGS WHERE patientId = ? ORDER BY timestamp ASC`, [patientId], (err, logs) => {
      const formattedLogs = logs.map(l => ({ ...l, forensicPayload: JSON.parse(l.forensicPayload) }));
      res.status(200).json({ success: true, profile: { ...profile, allergies: JSON.parse(profile.allergies || '[]') }, forensicTimeline: formattedLogs });
    });
  });
});
