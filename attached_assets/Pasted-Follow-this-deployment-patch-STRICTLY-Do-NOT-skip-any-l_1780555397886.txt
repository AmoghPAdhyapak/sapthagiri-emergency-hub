// Centralized Patient Master Folder System Layout Architecture
const MasterPatientsFolder = {
  registry: new Map(),

  mutatePatientStatus: function(patientId, targetStatus, verdictTier, reasoningPayload) {
    const patientFile = this.registry.get(patientId);
    if (!patientFile) return false;

    patientFile.currentEcosystemStatus = targetStatus;
    patientFile.symptomFinalVerdict = verdictTier;

    const historicalTimelineNode = {
      updateId: `STATUS-CHG-${Date.now()}`,
      timestamp: new Date().toLocaleDateString('en-GB', { day: 'numeric', month: 'short', year: 'numeric', hour: '2-digit', minute: '2-digit' }),
      reasonForStatusChange: reasoningPayload.reason,
      updatedSymptomsPresentation: reasoningPayload.updatedSymptoms,
      clinicalObservationNotes: reasoningPayload.observation,
      authorizingDoctorId: reasoningPayload.doctorId
    };

    patientFile.medicalTimelineHistory.push(historicalTimelineNode);
    this.registry.set(patientId, patientFile);

    localStorage.setItem("MASTER_PATIENTS_FOLDER_DATA", JSON.stringify(Array.from(this.registry.entries())));
    return true;
  }
};
